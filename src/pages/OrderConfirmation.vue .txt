<script setup lang="ts">
import { ref, onMounted } from 'vue'
import { useRoute, useRouter } from 'vue-router'
import { useOrdersStore } from '@/stores/orders'
import { useAuthStore } from '@/stores/auth'
import { useLanguageStore } from '@/stores/language'
import { authNotification } from '@/utils/notifications'

const route = useRoute()
const router = useRouter()
const ordersStore = useOrdersStore()
const authStore = useAuthStore()
const { t } = useLanguageStore()

const order = ref<any>(null)
const loading = ref(true)
const error = ref<string | null>(null)

const formatCurrency = (amount: number) => {
  return new Intl.NumberFormat('en-EG').format(amount)
}

const formatDate = (date: Date) => {
  return new Date(date).toLocaleDateString('en-EG', {
    year: 'numeric',
    month: 'long',
    day: 'numeric',
    hour: '2-digit',
    minute: '2-digit'
  })
}

const getStatusClasses = (status: string) => {
  const classes = {
    pending: 'bg-yellow-100 text-yellow-800',
    processing: 'bg-blue-100 text-blue-800',
    shipped: 'bg-purple-100 text-purple-800',
    delivered: 'bg-green-100 text-green-800',
    cancelled: 'bg-red-100 text-red-800'
  }
  return classes[status as keyof typeof classes] || 'bg-gray-100 text-gray-800'
}

const getStatusText = (status: string) => {
  const texts = {
    pending: t('Pending'),
    processing: t('Processing'),
    shipped: t('Shipped'),
    delivered: t('Delivered'),
    cancelled: t('Cancelled')
  }
  return texts[status as keyof typeof texts] || status
}

const getPaymentMethodText = (method: string) => {
  const methods: Record<string, string> = {
    cash_on_delivery: t('Cash on Delivery'),
    credit_card: t('Credit Card'),
    bank_transfer: t('Bank Transfer')
  }
  return methods[method] || method
}

const getPaymentStatusClasses = (status: string) => {
  const classes = {
    pending: 'bg-yellow-100 text-yellow-800',
    paid: 'bg-green-100 text-green-800',
    failed: 'bg-red-100 text-red-800',
    refunded: 'bg-gray-100 text-gray-800'
  }
  return classes[status as keyof typeof classes] || 'bg-gray-100 text-gray-800'
}

const loadOrder = async () => {
  const orderId = route.params.id as string
  
  if (!orderId) {
    error.value = t('Order ID not found')
    loading.value = false
    return
  }

  loading.value = true
  error.value = null

  try {
    // Try to get from store first
    let fetchedOrder = ordersStore.currentOrder
    
    // If not in store, fetch from Firestore
    if (!fetchedOrder || fetchedOrder.id !== orderId) {
      fetchedOrder = await ordersStore.fetchOrderById(orderId)
    }

    if (fetchedOrder) {
      order.value = fetchedOrder
      
      // Save guest info to localStorage for future reference
      if (fetchedOrder.guestId) {
        localStorage.setItem('guest_order_id', fetchedOrder.guestId)
        localStorage.setItem('last_order_email', fetchedOrder.customer.email)
        localStorage.setItem('last_order_number', fetchedOrder.orderNumber)
      }
    } else {
      error.value = t('Order not found')
    }
  } catch (err: any) {
    console.error('Error loading order:', err)
    
    // Handle permission errors gracefully
    if (err.code === 'permission-denied') {
      error.value = t('Unable to access order. Please check your email for order details.')
    } else {
      error.value = t('Failed to load order details')
    }
  } finally {
    loading.value = false
  }
}

const retryLoadOrder = () => {
  loadOrder()
}

const downloadInvoice = () => {
  if (order.value) {
    ordersStore.downloadInvoice(order.value.id)
  }
}

const copyTrackLink = () => {
  const link = `${window.location.origin}/track-order?order=${order.value?.orderNumber}&email=${order.value?.customer.email}`
  navigator.clipboard.writeText(link)
  authNotification.loggedIn(t('Track link copied to clipboard'))
}

onMounted(() => {
  loadOrder()
})
</script>
